🔧 Supabase 연동 설정

아래 SQL을 실행한 후 URL과 Key를 입력하세요.
1
Supabase 프로젝트 생성
supabase.com 무료 가입 → New Project
2
Authentication → Email → Confirm email OFF
이메일 인증을 끄면 가입 즉시 로그인됩니다
3
SQL Editor에서 아래 SQL 실행
-- 프로필 (역할: teacher / student, 반코드: "2-3") create table profiles ( id uuid references auth.users primary key, name text not null, class_code text not null, role text not null default 'student', created_at timestamptz default now() ); alter table profiles enable row level security; create policy "read all profiles" on profiles for select using (true); create policy "own profile write" on profiles for all using (auth.uid() = id); -- 수행평가 (class_code 기준 공유) create table performances ( id uuid primary key default gen_random_uuid(), class_code text not null, created_by uuid references auth.users not null, subject text, type text, title text, description text, date date, period text, color text, created_at timestamptz default now() ); alter table performances enable row level security; create policy "class read" on performances for select using (class_code = (select class_code from profiles where id = auth.uid())); create policy "teacher write" on performances for insert using ((select role from profiles where id = auth.uid()) = 'teacher'); create policy "teacher delete" on performances for delete using (created_by = auth.uid()); -- 시험정보 (class_code 기준 공유) create table exams ( id uuid primary key default gen_random_uuid(), class_code text not null, created_by uuid references auth.users not null, subject text, exam_date date, start_time text, range_text text, written_ratio int, performance_ratio int, attendance_ratio int, created_at timestamptz default now() ); alter table exams enable row level security; create policy "class read" on exams for select using (class_code = (select class_code from profiles where id = auth.uid())); create policy "teacher write" on exams for insert using ((select role from profiles where id = auth.uid()) = 'teacher'); create policy "teacher delete" on exams for delete using (created_by = auth.uid()); -- 시간표 (class_code 기준 공유) create table timetables ( id uuid primary key default gen_random_uuid(), class_code text not null, created_by uuid references auth.users not null, period int, mon text, tue text, wed text, thu text, fri text, unique(class_code, period) ); alter table timetables enable row level security; create policy "class read" on timetables for select using (class_code = (select class_code from profiles where id = auth.uid())); create policy "teacher write" on timetables for all using ((select role from profiles where id = auth.uid()) = 'teacher');
